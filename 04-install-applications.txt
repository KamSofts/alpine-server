INSTALL MARIADB SERVER:-
~~~~~~~~~~~~~~~~~~~~~~~~
    ~# apk update
    ~# apk add mariadb mariadb-client mariadb-openrc
    ~# mariadb-install-db --user=mysql --datadir=/var/lib/mysql
    ~# rc-service mariadb start
    ~# rc-update add mariadb default
    ~# mysql_secure_installation 
        Current root password       : enter
        unix_socket authendication  : n 
        Change root password        : n 
        Remove anonymous user       : Y 
        Disallow root login remotely: Y 
        Remove test database        : Y 
        Reload privilege tables now : Y 
    
    ~# mariadb
        > CREATE DATABASE ram_db;
        > CREATE USER ram@`%` IDENTIFIED BY 'ram';
        > GRANT ALL PRIVILEGES ON ram_db.* TO ram@`%`;
        > FLUSH PRIVILEGES;
        > exit;

REMOTE ACCESS FOR MARIADB:-
~~~~~~~~~~~~~~~~~~~~~~~~~~~
    ~# nano /etc/my.cfn.d/mariadb_server.cnf
        [mysqld]
        #skip-networking
        bind-address=0.0.0.0    
        // save and exit from mariadb_server.cnf
    ~# rc-service mariadb restart
